pragma solidity ^0.8.20;

import "@openzeppelin/contracts/token/ERC721/ERC721.sol";
import "@openzeppelin/contracts/access/Ownable.sol";
import "@openzeppelin/contracts/utils/Strings.sol";

contract CharapiaNft is ERC721, Ownable {
    uint256 public nextTokenId = 1;
    address public marketplaceAddress;

    struct Character {
        string name;
        string image;
    }

    mapping(uint256 => Character) public characters;

    constructor(
        address _marketplaceAddress
    ) ERC721("Charapia AI Character", "CHAI") {
        marketplaceAddress = _marketplaceAddress;
    }

    function mint(string memory name_, string memory image_) external {
        uint256 tokenId = nextTokenId;
        nextTokenId++;

        _safeMint(msg.sender, tokenId);
        characters[tokenId] = Character(name_, image_);

        _setApprovalForAll(msg.sender, marketplaceAddress, true);
    }

    function getCharacter(
        uint256 tokenId
    ) external view returns (string memory name, string memory image) {
        require(tokenId > 0 && tokenId < nextTokenId, "NFT does not exist");
        Character memory character = characters[tokenId];
        return (character.name, character.image);
    }

    function setMarketplace(address _marketplaceAddress) external onlyOwner {
        marketplaceAddress = _marketplaceAddress;
    }

    // Fixed tokenURI
    function tokenURI(
        uint256 tokenId
    ) public view override returns (string memory) {
        // In v5, checking ownerOf returns address(0) if it doesn't exist
        require(
            _exists(tokenId),
            "ERC721Metadata: URI query for nonexistent token"
        );

        Character memory character = characters[tokenId];
        string memory json = string(
            abi.encodePacked(
                '{"name":"',
                character.name,
                '", "description":"Charapia AI Character NFT",',
                ' "image":"',
                character.image,
                '"}'
            )
        );

        return
            string(
                abi.encodePacked(
                    "data:application/json;base64,",
                    Base64.encode(bytes(json))
                )
            );
    }
}

/// @notice Minimal Base64 library for encoding
library Base64 {
    bytes internal constant TABLE =
        "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/";

    function encode(bytes memory data) internal pure returns (string memory) {
        if (data.length == 0) return "";

        uint256 encodedLen = 4 * ((data.length + 2) / 3);
        bytes memory result = new bytes(encodedLen + 32);

        bytes memory table = TABLE;

        assembly {
            let tablePtr := add(table, 1)
            let resultPtr := add(result, 32)
            for {
                let i := 0
            } lt(i, mload(data)) {} {
                i := add(i, 3)
                let input := and(mload(add(data, i)), 0xffffff)
                let out := mload(add(tablePtr, and(shr(18, input), 0x3F)))
                out := shl(8, out)
                out := add(out, mload(add(tablePtr, and(shr(12, input), 0x3F))))
                out := shl(8, out)
                out := add(out, mload(add(tablePtr, and(shr(6, input), 0x3F))))
                out := shl(8, out)
                out := add(out, mload(add(tablePtr, and(input, 0x3F))))
                out := shl(224, out)
                mstore(resultPtr, out)
                resultPtr := add(resultPtr, 4)
            }
            mstore(result, encodedLen)
        }

        return string(result);
    }
}
